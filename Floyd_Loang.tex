\section{Thuật toán Floyd-Warshall}

\begin{frame}{Khái niệm Floyd-Warshall}
\begin{itemize}
    \item \textbf{Mục tiêu:} Tìm đường đi ngắn nhất giữa mọi cặp đỉnh $(u, v)$ trong đồ thị.
    \item \textbf{Đặc điểm:} 
    \begin{itemize}
        \item Hoạt động trên đồ thị có hướng/vô hướng.
        \item Chấp nhận trọng số cạnh âm.
        \item Không hoạt động nếu đồ thị có chu trình âm.
    \end{itemize}
    \item \textbf{Ý tưởng:} Duyệt qua từng đỉnh trung gian $k$ để cập nhật khoảng cách giữa mọi cặp $(i, j)$.
\end{itemize}
\end{frame}

\begin{frame}{Công thức và Độ phức tạp}
\begin{block}{Công thức truy hồi}
Gọi $d[i][j]$ là khoảng cách ngắn nhất từ $i$ đến $j$:
\[
d[i][j] = \min(d[i][j], d[i][k] + d[k][j])
\]
Với $k$ là đỉnh trung gian đang xét.
\end{block}

\begin{itemize}
    \item \textbf{Độ phức tạp thời gian:} $O(V^3)$ (3 vòng lặp lồng nhau).
    \item \textbf{Độ phức tạp không gian:} $O(V^2)$ (Ma trận kề).
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Cài đặt Floyd-Warshall (C++)}
\begin{lstlisting}
// Khoi tao: d[i][j] = trong so canh, d[i][i] = 0, con lai la INF
void floydWarshall(int n) {
    for (int k = 1; k <= n; ++k) { // Dinh trung gian k
        for (int i = 1; i <= n; ++i) { // Dinh dau i
            for (int j = 1; j <= n; ++j) { // Dinh cuoi j
                if (d[i][k] < INF && d[k][j] < INF) {
                    d[i][j] = min(d[i][j], d[i][k] + d[k][j]);
                }
            }
        }
    }
}
\end{lstlisting}
\end{frame}

\section{Thuật toán Loang trên bảng}

\begin{frame}{Khái niệm Loang (Flood Fill/BFS)}
\begin{itemize}
    \item \textbf{Phạm vi:} Thường áp dụng trên ma trận (lưới ô vuông) 2 chiều.
    \item \textbf{Ứng dụng:} 
    \begin{itemize}
        \item Tìm đường đi ngắn nhất trên lưới không trọng số.
        \item Đếm số lượng thành phần liên thông (vùng màu, vùng đất...).
        \item Xác định các ô có thể đi tới từ một vị trí gốc.
    \end{itemize}
    \item \textbf{Kỹ thuật:} Sử dụng mảng hướng (Direction arrays) để di chuyển giữa các ô kề cạnh.
\end{itemize}
\end{frame}

\textbf{Mảng hướng (4 hướng):}
\begin{itemize}
    \item \texttt{dx[] = \{-1, 1, 0, 0\}}
    \item \texttt{dy[] = \{0, 0, -1, 1\}}
\end{itemize}
\textbf{Tọa độ mới:}
\[ nx = x + dx[i], ny = y + dy[i] \]
\end{columns}
\end{frame}

\begin{frame}[fragile]{Cài đặt Loang bằng BFS (C++)}
\begin{lstlisting}
void bfs_loang(int r, int c) {
    queue<pair<int, int>> q;
    q.push({r, c});
    visited[r][c] = true;
    dist[r][c] = 0;
    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();
        for (int i = 0; i < 4; i++) {
            int nx = x + dx[i], ny = y + dy[i];
            if (nx >= 1 && nx <= N && ny >= 1 && ny <= M && !visited[nx][ny] && a[nx][ny] != '#') {
                visited[nx][ny] = true;
                dist[nx][ny] = dist[x][y] + 1;
                q.push({nx, ny});
            }
        }
    }
}
\end{lstlisting}
\end{frame}
