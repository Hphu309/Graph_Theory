void floydWarshall(int n) {
    for (int k = 1; k <= n; ++k) { // Dinh trung gian k
        for (int i = 1; i <= n; ++i) { // Dinh dau i
            for (int j = 1; j <= n; ++j) { // Dinh cuoi j
                if (d[i][k] < INF && d[k][j] < INF) {
                    d[i][j] = min(d[i][j], d[i][k] + d[k][j]);
                }
            }
        }
    }
}
